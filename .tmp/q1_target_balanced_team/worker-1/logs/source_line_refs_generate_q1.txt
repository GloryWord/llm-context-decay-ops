   865	def write_trace_csv(cases: list[dict], output_path: Path, max_turns: int = TRACE_CSV_MAX_TURNS) -> None:
   866	    """Write a one-row-per-case provenance CSV for thesis/design inspection."""
   867	    output_path.parent.mkdir(parents=True, exist_ok=True)
   868	    fieldnames = trace_csv_fieldnames(max_turns)
   869	    with output_path.open("w", encoding="utf-8-sig", newline="") as f:
   870	        writer = csv.DictWriter(f, fieldnames=fieldnames)
   871	        writer.writeheader()
   872	        for case in cases:
   873	            writer.writerow(flatten_case_for_trace_csv(case, max_turns=max_turns))
